SELECT user.id AS user_id, user.email AS email, user.name AS name, user.asal_sekolah, user.date_created, user.is_active AS active, user.reff AS reff, user.gender AS jenis_kelamin, program.kode_program AS program, user.kip AS kip, user.village_id AS desa, data_wilayah_new.nm_wil AS kecamatan, data_kabupaten.nm_wil AS kabupaten, data_provinsi.nm_wil AS provinsi FROM user JOIN kelas ON user.kelas_id = kelas.kelas_id JOIN program ON kelas.prog_id = program.prog_id JOIN data_wilayah_new ON user.district_id = data_wilayah_new.id_wil JOIN data_kabupaten ON user.regency_id = data_kabupaten.id_wil JOIN data_provinsi ON user.province_id = data_provinsi.id_wil ORDER BY user.id ASC;
